\doxysubsection{Detailed Description}
Define a set of one or more 2D primitives. 

\mbox{\hyperlink{classsf_1_1_vertex_array}{sf\+::\+Vertex\+Array}} is a very simple wrapper around a dynamic array of vertices and a primitives type.

It inherits \mbox{\hyperlink{classsf_1_1_drawable}{sf\+::\+Drawable}}, but unlike other drawables it is not transformable.

Example\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_vertex_array}{sf::VertexArray}} lines(\mbox{\hyperlink{group__graphics_gga5ee56ac1339984909610713096283b1baa58596ca614193671ca68cc213d8ecf1}{sf::LineStrip}}, 4);}
\DoxyCodeLine{lines[0].position = \mbox{\hyperlink{namespacesf_a80cea3c46537294fd1d8d428566ad8b2}{sf::Vector2f}}(10, 0);}
\DoxyCodeLine{lines[1].position = \mbox{\hyperlink{namespacesf_a80cea3c46537294fd1d8d428566ad8b2}{sf::Vector2f}}(20, 0);}
\DoxyCodeLine{lines[2].position = \mbox{\hyperlink{namespacesf_a80cea3c46537294fd1d8d428566ad8b2}{sf::Vector2f}}(30, 5);}
\DoxyCodeLine{lines[3].position = \mbox{\hyperlink{namespacesf_a80cea3c46537294fd1d8d428566ad8b2}{sf::Vector2f}}(40, 2);}
\DoxyCodeLine{}
\DoxyCodeLine{window.draw(lines);}
