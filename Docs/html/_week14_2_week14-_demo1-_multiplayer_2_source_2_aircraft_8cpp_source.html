109void Aircraft::updateCurrent(sf::Time dt, CommandQueue& commands)
110{
111 // Update texts and roll animation
112 updateTexts();
113 updateRollAnimation();
114
115 // Entity has been destroyed: Possibly drop pickup, mark for removal
116 if (isDestroyed())
117 {
118 checkPickupDrop(commands);
119 mExplosion.update(dt);
120
121 // Play explosion sound only once
122 if (!mExplosionBegan)
123 {
124 // Play sound effect
125 SoundEffect::ID soundEffect = (randomInt(2) == 0) ? SoundEffect::Explosion1 : SoundEffect::Explosion2;
126 playLocalSound(commands, soundEffect);
127
128 // Emit network game action for enemy explosions
129 if (!isAllied())
130 {
131 sf::Vector2f position = getWorldPosition();
132
133 Command command;
134 command.category = Category::Network;
135 command.action = derivedAction<NetworkNode>([position](NetworkNode& node, sf::Time)
136 {
137 node.notifyGameAction(GameActions::EnemyExplode, position);
138 });
139
140 commands.push(command);
141 }
142
143 mExplosionBegan = true;
144 }
145 return;
146 }
147
148 // Check if bullets or missiles are fired
149 checkProjectileLaunch(dt, commands);
150
151 // Update enemy movement pattern; apply velocity
152 updateMovementPattern(dt);
153 Entity::updateCurrent(dt, commands);
154}

248void Aircraft::updateMovementPattern(sf::Time dt)
249{
250 // Enemy airplane: Movement pattern
251 const std::vector<Direction>& directions = Table[mType].directions;
252 if (!directions.empty())
253 {
254 // Moved long enough in current direction: Change direction
255 if (mTravelledDistance > directions[mDirectionIndex].distance)
256 {
257 mDirectionIndex = (mDirectionIndex + 1) % directions.size();
258 mTravelledDistance = 0.f;
259 }
260
261 // Compute velocity from direction
262 float radians = toRadian(directions[mDirectionIndex].angle + 90.f);
263 float vx = getMaxSpeed() * std::cos(radians);
264 float vy = getMaxSpeed() * std::sin(radians);
265
266 setVelocity(vx, vy);
267
268 mTravelledDistance += getMaxSpeed() * dt.asSeconds();
269 }
270}

282void Aircraft::checkProjectileLaunch(sf::Time dt, CommandQueue& commands)
283{
284 // Enemies try to fire all the time
285 if (!isAllied())
286 fire();
287
288 // Check for automatic gunfire, allow only in intervals
289 if (mIsFiring && mFireCountdown <= sf::Time::Zero)
290 {
291 // Interval expired: We can fire a new bullet
292 commands.push(mFireCommand);
293 playLocalSound(commands, isAllied() ? SoundEffect::AlliedGunfire : SoundEffect::EnemyGunfire);
294
295 mFireCountdown += Table[mType].fireInterval / (mFireRateLevel + 1.f);
296 mIsFiring = false;
297 }
298 else if (mFireCountdown > sf::Time::Zero)
299 {
300 // Interval not expired: Decrease it further
301 mFireCountdown -= dt;
302 mIsFiring = false;
303 }
304
305 // Check for missile launch
306 if (mIsLaunchingMissile)
307 {
308 commands.push(mMissileCommand);
309 playLocalSound(commands, SoundEffect::LaunchMissile);
310
311 mIsLaunchingMissile = false;
312 }
