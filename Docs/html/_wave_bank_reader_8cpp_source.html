103 WORD BitsPerSample() const noexcept
104 {
105 if (wFormatTag == TAG_XMA)
106 return 16; // XMA_OUTPUT_SAMPLE_BITS == 16
107 if (wFormatTag == TAG_WMA)
108 return 16;
109 if (wFormatTag == TAG_ADPCM)
110 return 4; // MSADPCM_BITS_PER_SAMPLE == 4
111
112 // wFormatTag must be TAG_PCM (2 bits can only represent 4 different values)
113 return (wBitsPerSample == BITDEPTH_16) ? 16u : 8u;
114 }
115
116 DWORD BlockAlign() const noexcept
117 {
118 switch (wFormatTag)
119 {
120 case TAG_PCM:
121 return wBlockAlign;
122
123 case TAG_XMA:
124 return (nChannels * 16 / 8); // XMA_OUTPUT_SAMPLE_BITS = 16
125
126 case TAG_ADPCM:
127 return (wBlockAlign + ADPCM_BLOCKALIGN_CONVERSION_OFFSET) * nChannels;
128
129 case TAG_WMA:
130 {
131 static const uint32_t aWMABlockAlign[] =
132 {
133 929,
134 1487,
135 1280,
136 2230,
137 8917,
138 8192,
139 4459,
140 5945,
141 2304,
142 1536,
143 1485,
144 1008,
145 2731,
146 4096,
147 6827,
148 5462,
149 1280
150 };
151
152 uint32_t dwBlockAlignIndex = wBlockAlign & 0x1F;
153 if (dwBlockAlignIndex < _countof(aWMABlockAlign))
154 return aWMABlockAlign[dwBlockAlignIndex];
155 }
156 break;
157 }
158
159 return 0;
160 }
161
162 DWORD AvgBytesPerSec() const noexcept
163 {
164 switch (wFormatTag)
165 {
166 case TAG_PCM:
167 return nSamplesPerSec * wBlockAlign;
168
169 case TAG_XMA:
170 return nSamplesPerSec * BlockAlign();
171
172 case TAG_ADPCM:
173 {
174 uint32_t blockAlign = BlockAlign();
175 uint32_t samplesPerAdpcmBlock = AdpcmSamplesPerBlock();
176 return blockAlign * nSamplesPerSec / samplesPerAdpcmBlock;
177 }
178
179 case TAG_WMA:
180 {
181 static const uint32_t aWMAAvgBytesPerSec[] =
182 {
183 12000,
184 24000,
185 4000,
186 6000,
187 8000,
188 20000,
189 2500
190 };
191 // bitrate = entry * 8
192
193 uint32_t dwBytesPerSecIndex = wBlockAlign >> 5;
194 if (dwBytesPerSecIndex < _countof(aWMAAvgBytesPerSec))
195 return aWMAAvgBytesPerSec[dwBytesPerSecIndex];
196 }
197 break;
198 }
199
200 return 0;
201 }

203 DWORD AdpcmSamplesPerBlock() const noexcept
204 {
205 uint32_t nBlockAlign = (wBlockAlign + ADPCM_BLOCKALIGN_CONVERSION_OFFSET) * nChannels;
206 return nBlockAlign * 2 / uint32_t(nChannels) - 12;

279 void ComputeLocations(DWORD& offset, DWORD& length, uint32_t index, const HEADER& header, const BANKDATA& data, const ENTRYCOMPACT* entries) const noexcept
280 {
281 offset = dwOffset * data.dwAlignment;
282
283 if (index < (data.dwEntryCount - 1))
284 {
285 length = (entries[index + 1].dwOffset * data.dwAlignment) - offset - dwLengthDeviation;
286 }
287 else
288 {
289 length = header.Segments[HEADER::SEGIDX_ENTRYWAVEDATA].dwLength - offset - dwLengthDeviation;

293 static uint32_t GetDuration(DWORD length, const BANKDATA& data, const uint32_t* seekTable) noexcept
294 {
295 switch (data.CompactFormat.wFormatTag)
296 {
297 case MINIWAVEFORMAT::TAG_ADPCM:
298 {
299 uint32_t duration = (length / data.CompactFormat.BlockAlign()) * data.CompactFormat.AdpcmSamplesPerBlock();
300 uint32_t partial = length % data.CompactFormat.BlockAlign();
301 if (partial)
302 {
303 if (partial >= (7u * data.CompactFormat.nChannels))
304 duration += (partial * 2 / data.CompactFormat.nChannels - 12);
305 }
306 return duration;
307 }
308
309 case MINIWAVEFORMAT::TAG_WMA:
310 if (seekTable)
311 {
312 uint32_t seekCount = *seekTable;
313 if (seekCount > 0)
314 {
315 return seekTable[seekCount] / uint32_t(2 * data.CompactFormat.nChannels);
316 }
317 }
318 return 0;
319
320 case MINIWAVEFORMAT::TAG_XMA:
321 if (seekTable)
322 {
323 uint32_t seekCount = *seekTable;
324 if (seekCount > 0)
325 {
326 return seekTable[seekCount];
327 }
328 }
329 return 0;
330
331 default:
332 return uint32_t((uint64_t(length) * 8)
333 / (uint64_t(data.CompactFormat.BitsPerSample()) * uint64_t(data.CompactFormat.nChannels)));
334 }
335 }

340 inline const uint32_t* FindSeekTable(uint32_t index, const uint8_t* seekTable, const HEADER& header, const BANKDATA& data) noexcept
341 {
342 if (!seekTable || index >= data.dwEntryCount)
343 return nullptr;
344
345 uint32_t seekSize = header.Segments[HEADER::SEGIDX_SEEKTABLES].dwLength;
346
347 if ((index * sizeof(uint32_t)) > seekSize)
348 return nullptr;
349
350 auto table = reinterpret_cast<const uint32_t*>(seekTable);
351 uint32_t offset = table[index];
352 if (offset == uint32_t(-1))
353 return nullptr;
354
355 offset += sizeof(uint32_t) * data.dwEntryCount;
356
357 if (offset > seekSize)
358 return nullptr;
359
360 return reinterpret_cast<const uint32_t*>(seekTable + offset);
361 }
362}

729 if (m_data.dwFlags & BANKDATA::TYPE_STREAMING)
730 {
731 // If streaming, reopen without buffering
732 hFile.reset();
733
734 #if (_WIN32_WINNT >= _WIN32_WINNT_WIN8)
735 CREATEFILE2_EXTENDED_PARAMETERS params2 = { sizeof(CREATEFILE2_EXTENDED_PARAMETERS), 0, 0, 0, {}, nullptr };
736 params2.dwFileAttributes = FILE_ATTRIBUTE_NORMAL;
737 params2.dwFileFlags = FILE_FLAG_OVERLAPPED | FILE_FLAG_NO_BUFFERING;
738 m_async = CreateFile2(szFileName,
739 GENERIC_READ,
740 FILE_SHARE_READ,
741 OPEN_EXISTING,
742 &params2);
743 #else
744 m_async = CreateFileW(szFileName,
745 GENERIC_READ,
746 FILE_SHARE_READ,
747 nullptr,
748 OPEN_EXISTING,
749 FILE_FLAG_OVERLAPPED | FILE_FLAG_NO_BUFFERING,
750 nullptr);
751 #endif
752
753 if (m_async == INVALID_HANDLE_VALUE)
754 {
755 return HRESULT_FROM_WIN32(GetLastError());
756 }
757
758 m_prepared = true;
759 }
760 else
761 {
762 // If in-memory, kick off read of wave data
763 void* dest = nullptr;
764
765 {
766 m_waveData.reset(new (std::nothrow) uint8_t[waveLen]);
767 if (!m_waveData)
768 return E_OUTOFMEMORY;
769
770 dest = m_waveData.get();
771 }
772
773 memset(&m_request, 0, sizeof(OVERLAPPED));
774 m_request.Offset = m_header.Segments[HEADER::SEGIDX_ENTRYWAVEDATA].dwOffset;
775 m_request.hEvent = m_event.get();
776
777 if (!ReadFile(hFile.get(), dest, waveLen, nullptr, &m_request))
778 {
779 DWORD error = GetLastError();
780 if (error != ERROR_IO_PENDING)
781 return HRESULT_FROM_WIN32(error);
782 }
783 else
784 {
785 m_prepared = true;
786 memset(&m_request, 0, sizeof(OVERLAPPED));
787 }
788
789 m_async = hFile.release();
790 }
791
792 return S_OK;
793}

820HRESULT WaveBankReader::Impl::GetFormat(uint32_t index, WAVEFORMATEX* pFormat, size_t maxsize) const noexcept
821{
822 if (!pFormat || !maxsize)
823 return E_INVALIDARG;
824
825 if (index >= m_data.dwEntryCount || !m_entries)
826 {
827 return E_FAIL;
828 }
829
830 auto& miniFmt = (m_data.dwFlags & BANKDATA::FLAGS_COMPACT) ? m_data.CompactFormat : (reinterpret_cast<const ENTRY*>(m_entries.get())[index].Format);
831
832 switch (miniFmt.wFormatTag)
833 {
834 case MINIWAVEFORMAT::TAG_PCM:
835 if (maxsize < sizeof(PCMWAVEFORMAT))
836 return HRESULT_FROM_WIN32(ERROR_MORE_DATA);
837
838 pFormat->wFormatTag = WAVE_FORMAT_PCM;
839
840 if (maxsize >= sizeof(WAVEFORMATEX))
841 {
842 pFormat->cbSize = 0;
843 }
844 break;
845
846 case MINIWAVEFORMAT::TAG_ADPCM:
847 if (maxsize < (sizeof(WAVEFORMATEX) + 32 /*MSADPCM_FORMAT_EXTRA_BYTES*/))
848 return HRESULT_FROM_WIN32(ERROR_MORE_DATA);
849
850 pFormat->wFormatTag = WAVE_FORMAT_ADPCM;
851 pFormat->cbSize = 32 /*MSADPCM_FORMAT_EXTRA_BYTES*/;
852 {
853 auto adpcmFmt = reinterpret_cast<ADPCMWAVEFORMAT*>(pFormat);
854 adpcmFmt->wSamplesPerBlock = static_cast<WORD>(miniFmt.AdpcmSamplesPerBlock());
855 miniFmt.AdpcmFillCoefficientTable(adpcmFmt);
856 }
857 break;
858
859 case MINIWAVEFORMAT::TAG_WMA:
860 if (maxsize < sizeof(WAVEFORMATEX))
861 return HRESULT_FROM_WIN32(ERROR_MORE_DATA);
862
863 pFormat->wFormatTag = static_cast<WORD>((miniFmt.wBitsPerSample & 0x1) ? WAVE_FORMAT_WMAUDIO3 : WAVE_FORMAT_WMAUDIO2);
864 pFormat->cbSize = 0;
865 break;
866
867 case MINIWAVEFORMAT::TAG_XMA: // XMA2 is supported by Xbox One
868 return HRESULT_FROM_WIN32(ERROR_NOT_SUPPORTED);
869
870 default:
871 return E_FAIL;
872 }
873
874 pFormat->nChannels = miniFmt.nChannels;
875 pFormat->wBitsPerSample = miniFmt.BitsPerSample();
876 pFormat->nBlockAlign = static_cast<WORD>(miniFmt.BlockAlign());
877 pFormat->nSamplesPerSec = miniFmt.nSamplesPerSec;
878 pFormat->nAvgBytesPerSec = miniFmt.AvgBytesPerSec();
879
880 return S_OK;
881}
