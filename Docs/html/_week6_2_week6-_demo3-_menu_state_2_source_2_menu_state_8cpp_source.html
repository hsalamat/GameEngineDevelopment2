54bool MenuState::handleEvent(const sf::Event& event)
55{
56 // The demonstration menu logic
57 if (event.type != sf::Event::KeyPressed)
58 return false;
59
60 if (event.key.code == sf::Keyboard::Return)
61 {
62 if (mOptionIndex == Play)
63 {
64 requestStackPop();
65 //requestStackPush(States::Game);
66 }
67 else if (mOptionIndex == Exit)
68 {
69 // The exit option was chosen, by removing itself, the stack will be empty, and the game will know it is time to close.
70 requestStackPop();
71 }
72 }
73
74 else if (event.key.code == sf::Keyboard::Up)
75 {
76 // Decrement and wrap-around
77 if (mOptionIndex > 0)
78 mOptionIndex--;
79 else
80 mOptionIndex = mOptions.size() - 1;
81
82 updateOptionText();
83 }
84
85 else if (event.key.code == sf::Keyboard::Down)
86 {
87 // Increment and wrap-around
88 if (mOptionIndex < mOptions.size() - 1)
89 mOptionIndex++;
90 else
91 mOptionIndex = 0;
92
93 updateOptionText();
94 }
95
96 return true;
97}
