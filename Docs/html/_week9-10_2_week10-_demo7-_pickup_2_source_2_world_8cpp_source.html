209void World::guideMissiles()
210{
211 // Setup command that stores all enemies in mActiveEnemies
212 Command enemyCollector;
213 enemyCollector.category = Category::EnemyAircraft;
214 enemyCollector.action = derivedAction<Aircraft>([this](Aircraft& enemy, sf::Time)
215 {
216 if (!enemy.isDestroyed())
217 mActiveEnemies.push_back(&enemy);
218 });
219
220 // Setup command that guides all missiles to the enemy which is currently closest to the player
221 Command missileGuider;
222 missileGuider.category = Category::AlliedProjectile;
223 missileGuider.action = derivedAction<Projectile>([this](Projectile& missile, sf::Time)
224 {
225 // Ignore unguided bullets
226 if (!missile.isGuided())
227 return;
228
229 float minDistance = std::numeric_limits<float>::max();
230 Aircraft* closestEnemy = nullptr;
231
232 // Find closest enemy
233 for (Aircraft* enemy : mActiveEnemies)
234 {
235 float enemyDistance = distance(missile, *enemy);
236
237 if (enemyDistance < minDistance)
238 {
239 closestEnemy = enemy;
240 minDistance = enemyDistance;
241 }
242 }
243
244 if (closestEnemy)
245 missile.guideTowards(closestEnemy->getWorldPosition());
246 });
247
248 // Push commands, reset active enemies
249 mCommandQueue.push(enemyCollector);
250 mCommandQueue.push(missileGuider);
251 mActiveEnemies.clear();
252}
