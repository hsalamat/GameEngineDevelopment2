99void World::adaptPlayerPosition()
100{
101 // Keep player's position inside the screen bounds, at least borderDistance units from the border
102 sf::FloatRect viewBounds(mWorldView.getCenter() - mWorldView.getSize() / 2.f, mWorldView.getSize());
103 const float borderDistance = 100.f;
104
105 sf::Vector2f position = mPlayerAircraft->getPosition();
106 position.x = std::max(position.x, viewBounds.left + borderDistance);
107 position.x = std::min(position.x, viewBounds.left + viewBounds.width - borderDistance);
108 position.y = std::max(position.y, viewBounds.top + borderDistance);
109 position.y = std::min(position.y, viewBounds.top + viewBounds.height - borderDistance);
110 mPlayerAircraft->setPosition(position);
