33void World::update(sf::Time dt)
34{
35 // Scroll the world, reset player velocity
36 mWorldView.move(0.f, mScrollSpeed * dt.asSeconds());
37 mPlayerAircraft->setVelocity(0.f, 0.f);
38
39 // Setup commands to destroy entities, and guide missiles
40 destroyEntitiesOutsideView();
41 guideMissiles();
42
43 // Forward commands to scene graph, adapt velocity (scrolling, diagonal correction)
44 while (!mCommandQueue.isEmpty())
45 mSceneGraph.onCommand(mCommandQueue.pop(), dt);
46 adaptPlayerVelocity();
47
48 // Collision detection and response (may destroy entities)
49 handleCollisions();
50
51 // Remove all destroyed entities, create new ones
52 mSceneGraph.removeWrecks();
53 spawnEnemies();
54
55 // Regular update step, adapt position (correct if outside view)
56 mSceneGraph.update(dt, mCommandQueue);
57 adaptPlayerPosition();
58}

97void World::adaptPlayerPosition()
98{
99 // Keep player's position inside the screen bounds, at least borderDistance units from the border
100 sf::FloatRect viewBounds = getViewBounds();
101 const float borderDistance = 40.f;
102
103 sf::Vector2f position = mPlayerAircraft->getPosition();
104 position.x = std::max(position.x, viewBounds.left + borderDistance);
105 position.x = std::min(position.x, viewBounds.left + viewBounds.width - borderDistance);
106 position.y = std::max(position.y, viewBounds.top + borderDistance);
107 position.y = std::min(position.y, viewBounds.top + viewBounds.height - borderDistance);
108 mPlayerAircraft->setPosition(position);

123bool matchesCategories(SceneNode::Pair& colliders, Category::Type type1, Category::Type type2)
124{
125 unsigned int category1 = colliders.first->getCategory();
126 unsigned int category2 = colliders.second->getCategory();
127
128 // Make sure first pair entry has category type1 and second has type2
129 if (type1 & category1 && type2 & category2)
130 {
131 return true;
132 }
133 else if (type1 & category2 && type2 & category1)
134 {
135 std::swap(colliders.first, colliders.second);
136 return true;
137 }
138 else
139 {
140 return false;
141 }
142}

144void World::handleCollisions()
145{
146 std::set<SceneNode::Pair> collisionPairs;
147 mSceneGraph.checkSceneCollision(mSceneGraph, collisionPairs);
148
149 for(SceneNode::Pair pair: collisionPairs)
150 {
151 if (matchesCategories(pair, Category::PlayerAircraft, Category::EnemyAircraft))
152 {
153 auto& player = static_cast<Aircraft&>(*pair.first);
154 auto& enemy = static_cast<Aircraft&>(*pair.second);
155
156 // Collision: Player damage = enemy's remaining HP
157 player.damage(enemy.getHitpoints());
158 enemy.destroy();
159 }
160
161 else if (matchesCategories(pair, Category::PlayerAircraft, Category::Pickup))
162 {
163 auto& player = static_cast<Aircraft&>(*pair.first);
164 auto& pickup = static_cast<Pickup&>(*pair.second);
165
166 // Apply pickup effect to player, destroy projectile
167 pickup.apply(player);
168 pickup.destroy();
169 }
170
171 else if (matchesCategories(pair, Category::EnemyAircraft, Category::AlliedProjectile)
172 || matchesCategories(pair, Category::PlayerAircraft, Category::EnemyProjectile))
173 {
174 auto& aircraft = static_cast<Aircraft&>(*pair.first);
175 auto& projectile = static_cast<Projectile&>(*pair.second);
176
177 // Apply projectile damage to aircraft, destroy projectile
178 aircraft.damage(projectile.getDamage());
179 projectile.destroy();
180 }
181 }

274void World::guideMissiles()
275{
276 // Setup command that stores all enemies in mActiveEnemies
277 Command enemyCollector;
278 enemyCollector.category = Category::EnemyAircraft;
279 enemyCollector.action = derivedAction<Aircraft>([this] (Aircraft& enemy, sf::Time)
280 {
281 if (!enemy.isDestroyed())
282 mActiveEnemies.push_back(&enemy);
283 });
284
285 // Setup command that guides all missiles to the enemy which is currently closest to the player
286 Command missileGuider;
287 missileGuider.category = Category::AlliedProjectile;
288 missileGuider.action = derivedAction<Projectile>([this] (Projectile& missile, sf::Time)
289 {
290 // Ignore unguided bullets
291 if (!missile.isGuided())
292 return;
293
294 float minDistance = std::numeric_limits<float>::max();
295 Aircraft* closestEnemy = nullptr;
296
297 // Find closest enemy
298 for(Aircraft* enemy: mActiveEnemies)
299 {
300 float enemyDistance = distance(missile, *enemy);
301
302 if (enemyDistance < minDistance)
303 {
304 closestEnemy = enemy;
305 minDistance = enemyDistance;
306 }
307 }
308
309 if (closestEnemy)
310 missile.guideTowards(closestEnemy->getWorldPosition());
311 });
312
313 // Push commands, reset active enemies
314 mCommandQueue.push(enemyCollector);
315 mCommandQueue.push(missileGuider);
316 mActiveEnemies.clear();
317}
