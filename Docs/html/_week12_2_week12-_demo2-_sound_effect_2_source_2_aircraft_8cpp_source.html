177void Aircraft::updateMovementPattern(sf::Time dt)
178{
179 // Enemy airplane: Movement pattern
180 const std::vector<Direction>& directions = Table[mType].directions;
181 if (!directions.empty())
182 {
183 // Moved long enough in current direction: Change direction
184 if (mTravelledDistance > directions[mDirectionIndex].distance)
185 {
186 mDirectionIndex = (mDirectionIndex + 1) % directions.size();
187 mTravelledDistance = 0.f;
188 }
189
190 // Compute velocity from direction
191 float radians = toRadian(directions[mDirectionIndex].angle + 90.f);
192 float vx = getMaxSpeed() * std::cos(radians);
193 float vy = getMaxSpeed() * std::sin(radians);
194
195 setVelocity(vx, vy);
196
197 mTravelledDistance += getMaxSpeed() * dt.asSeconds();
198 }

209void Aircraft::checkProjectileLaunch(sf::Time dt, CommandQueue& commands)
210{
211 // Enemies try to fire all the time
212 if (!isAllied())
213 fire();
214
215 // Check for automatic gunfire, allow only in intervals
216 if (mIsFiring && mFireCountdown <= sf::Time::Zero)
217 {
218 // Interval expired: We can fire a new bullet
219 commands.push(mFireCommand);
220 mFireCountdown += Table[mType].fireInterval / (mFireRateLevel + 1.f);
221 mIsFiring = false;
222 }
223 else if (mFireCountdown > sf::Time::Zero)
224 {
225 // Interval not expired: Decrease it further
226 mFireCountdown -= dt;
227 mIsFiring = false;
228 }
229
230 // Check for missile launch
231 if (mIsLaunchingMissile)
232 {
233 commands.push(mMissileCommand);
234 mIsLaunchingMissile = false;
235 }
